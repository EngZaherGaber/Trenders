Introduction

Base URL: http://localhost:8000

This documentation aims to provide all the information you need to work with our API.

As you scroll, you'll see code examples for working with the API in different programming languages in the dark area to the right (or as part of the content on mobile). You can switch the language used with the tabs at the top right (or from the nav menu at the top left on mobile).

Authenticating requests

This API is not authenticated.

Authentication

POST api/login

Example request:
curl --request POST \
    "http://localhost:8000/api/login" \
    --header "Content-Type: application/json" \
    --header "Accept: application/json" \
    --data "{
    \"email\": \"test@test.com\",
    \"password\": \"password\"
}"

const url = new URL(
    "http://localhost:8000/api/login"
);

const headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
};

let body = {
    "email": "test@test.com",
    "password": "password"
};

fetch(url, {
    method: "POST",
    headers,
    body: JSON.stringify(body),
}).then(response => response.json());

POST api/register

Example request:
curl --request POST \
    "http://localhost:8000/api/register" \
    --header "Content-Type: multipart/form-data" \
    --header "Accept: application/json" \
    --form "name=quidem"\
    --form "email=daugherty.cora@example.com"\
    --form "password=password"\
    --form "type=instute"\
    --form "category_ids[]=1"\
    --form "created_in=2023-09-13T07:26:11"\
    --form "address=exercitationem"\
    --form "password_confirmation=password"\
    --form "image=@/tmp/phpu5z4fh" 
const url = new URL(
    "http://localhost:8000/api/register"
);

const headers = {
    "Content-Type": "multipart/form-data",
    "Accept": "application/json",
};

const body = new FormData();
body.append('name', 'quidem');
body.append('email', 'daugherty.cora@example.com');
body.append('password', 'password');
body.append('type', 'instute');
body.append('category_ids[]', '1');
body.append('created_in', '2023-09-13T07:26:11');
body.append('address', 'exercitationem');
body.append('password_confirmation', 'password');
body.append('image', document.querySelector('input[name="image"]').files[0]);

fetch(url, {
    method: "POST",
    headers,
    body,
}).then(response => response.json());
